99 -(void)updateDataInterface:(PxePlayerDataInterface *)interface finished:(isSuccess)success
100 {
101  if (interface.ncxURL || interface.masterPlaylist)
102  {
103  if(self.dataInterface) {
104  self.dataInterface = nil;
105  }
106 
107  self.dataInterface = interface;
108  self.dataInterface.searchIndexId = nil;
109  [[PxePlayerDataManager sharedInstance] setContextId:interface.contextId];
110 
111  if (interface.ncxURL)
112  {
113  NSString *lastPath = [[interface.ncxURL pathComponents] lastObject];
114 
115  self.dataInterface.baseURL = [interface.ncxURL stringByReplacingOccurrencesOfString:lastPath withString:@""];
116 
117  [self submitNCXUrlForIndexing:self.dataInterface.ncxURL withCompletionHandler:^(NSString *searchId, NSError *err)
118  {
119  if(!err)
120  {
121  DLog(@"Search Index ID: %@", searchId);
122  self.dataInterface.searchIndexId = searchId;
123  }
124  else {
125  self.dataInterface.searchIndexId = nil;
126  }
127  }];
128 
132 // interface.ncxURL = [interface.ncxURL stringByReplacingOccurrencesOfString:@"ncx" withString:@"xhtml"];
133  PxePlayerTocParser *tocParser = nil;
134  if(![[interface.ncxURL pathExtension] caseInsensitiveCompare:@"ncx"]) {
135  tocParser = [[PxePlayerNCXCDParser alloc] init];
136  }
137  else {
138  tocParser = [[PxePlayerXHTMLCDParser alloc] init];
139  }
140 
141  //download NCX content
142  [tocParser parseDataFromNCXURL:interface.ncxURL withHandler:^(id receivedNavigator, NSError *error)
143  {
144  if(!error) {
145  success([receivedNavigator boolValue]);
146  }
147  else {
148  success(NO);
149  }
150  }];
151  }
152  else
153  {
154  [self submitUrlsForIndexing:interface.masterPlaylist withCompletionHandler:^(NSString *searchId, NSError *err)
155  {
156  if(!err) {
157  self.dataInterface.searchIndexId = searchId;
158  }
159  else {
160  self.dataInterface.searchIndexId = nil;
161  }
162  }];
163  }
164  }
165 }
