45 -(BOOL)parsePageDetails:(NSDictionary*)data withParentId:(NSString*)parentId childrenFound:(BOOL)isChildren
46 {
47  NSDictionary *ncxContent = data[NCX_ANCHOR];
48  NSString *pageUrl = ncxContent[NCX_SRC];
49  if(pageUrl)
50  {
51  if([pageUrl rangeOfString:@"http:/"].location == NSNotFound) {
52  pageUrl = [[[PxePlayer sharedInstance] getNCXURL] stringByAppendingString:pageUrl];
53  }

57 
58  NSString *strippedURL = [pageUrl componentsSeparatedByString:@"#"][0];
59

61  [request setPredicate:[NSPredicate predicateWithFormat:@"(pageURL contains[cd] %@) AND (book.bookId = %@)", strippedURL, dataManager.contextId]];
62  [request setFetchLimit:1];
63 
64  NSInteger count = [[objectContext executeFetchRequest:request error:nil] count];
65  if(!count) {
66  ++self.pageNumber;

79  PxePlayerPageDetails *pageDetails = [NSEntityDescription insertNewObjectForEntityForName:NSStringFromClass([PxePlayerPageDetails class]) inManagedObjectContext:objectContext];

87  pageDetails.book = self.currentBook;
88 
89  if(![dataManager save]) {
90  return NO;
91  }
92  }
93  else {
94  return NO;
95  }
