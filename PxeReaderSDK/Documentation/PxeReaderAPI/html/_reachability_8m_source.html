191 {
192  PrintReachabilityFlags(flags, "localWiFiStatusForFlags");

202 - (NetworkStatus) networkStatusForFlags: (SCNetworkReachabilityFlags) flags
203 {
204  PrintReachabilityFlags(flags, "networkStatusForFlags");
205  if ((flags & kSCNetworkReachabilityFlagsReachable) == 0)
206  {
207  // if target host is not reachable
208  return NotReachable;
209  }
210 
211  BOOL retVal = NotReachable;
212 
213  if ((flags & kSCNetworkReachabilityFlagsConnectionRequired) == 0)
214  {
215  // if target host is reachable and no connection is required
216  // then we'll assume (for now) that your on Wi-Fi
217  retVal = ReachableViaWiFi;
218  }
219 
220 
221  if ((((flags & kSCNetworkReachabilityFlagsConnectionOnDemand ) != 0) ||
222  (flags & kSCNetworkReachabilityFlagsConnectionOnTraffic) != 0))
223  {
224  // ... and the connection is on-demand (or on-traffic) if the
225  // calling application is using the CFSocketStream or higher APIs
226 
227  if ((flags & kSCNetworkReachabilityFlagsInterventionRequired) == 0)
228  {
229  // ... and no [user] intervention is needed
230  retVal = ReachableViaWiFi;
231  }
232  }
233 
234  if ((flags & kSCNetworkReachabilityFlagsIsWWAN) == kSCNetworkReachabilityFlagsIsWWAN)
235  {
236  // ... but WWAN connections are OK if the calling application
237  // is using the CFNetwork (CFSocketStream?) APIs.
238  retVal = ReachableViaWWAN;
239  }
240  return retVal;
241 }

255 {
256  NSAssert(reachabilityRef != NULL, @"currentNetworkStatus called with NULL reachabilityRef");
257  NetworkStatus retVal = NotReachable;
258  SCNetworkReachabilityFlags flags;
259  if (SCNetworkReachabilityGetFlags(reachabilityRef, &flags))
